##2.1 变量的声明和初始化

fun main(args: Array<String>) {
    // val str = "Kotlin Programming language"          // 隐式声明
    val str: String = "Kotlin Programming language"      // 显式声明

    // val answer = 18
    val answer: Int = 18

    // 如果我们使用浮点常量作为变量的初始值，那么变量的类型是Double
    val fp = 12.3

    // 如果没有为变量赋予初始值，那么必须显式地指定变量的类型

}

##2.2 可变变量和不可变变量

fun main(args: Array<String>) {
    val a = 3
    val b = 5

    val str: String
    if (a > b) {
        str = "OK"
    } else {
        str = "Cancel"
    }

    val languages = arrayListOf("Java") // 声明一个不可变的引用
    languages.add("Kotlin") // 修改引用指向的对象

    // 编译器在推断变量的数据类型的时候，仅仅是通过变量的初始值进行推断
    // 而不会根据对变量初始化之后再给变量重新赋的值进行推断
    var i = 18
    // i = "hello"

    var相当于java中的final
}

##2.6 字符串模板

  1. 字符串是变量

    fun main(args: Array<String>) {

    val name = if (args.size > 0) args[0] else "Kotlin"

    // 相当于Java中的字符串连接："Hello, " + name + "!"
    println("Hello, $name!")

    // 如果字符串需要包含$，则需要进行转义处理
    println("Hello, \$name!")
}

    打印值：

    Hello，Kotlin!
Hello，$name!

  1. 字符串是表达式

    fun main(args: Array<String>) {

    // 占位符是表达式的时候，需要使用花括号{}
    if (args.size > 0) {
        println("Hello, ${args[0]}")
    }

    // 可以在双引号“”中嵌套双引号，但是这个双引号必须要在{}中
    println("Hello, ${if (args.size > 0) args[0] else "Kotlin"}")
}

##2.15 使用不带参数的when结构

import Color.*

enum class Color {
    RED, ORANGE, YELLOW, GREEN, BLUE, INDIGO, VIOLET
}

fun mix(c1: Color, c2: Color) =
        when (setOf(c1, c2)) {
            setOf(RED, YELLOW) -> ORANGE
            setOf(YELLOW, BLUE) -> GREEN
            setOf(BLUE, VIOLET) -> INDIGO
            else -> throw Exception("Dirty Color")
        }

fun mixOptimized(c1: Color, c2: Color) =
        when {
            (c1 == RED && c2 == YELLOW) ||
                    (c1 == YELLOW && c2 == RED)-> ORANGE
            (c1 == YELLOW && c2 == BLUE) ||
                    (c1 == BLUE && c2 == YELLOW) -> GREEN
            (c1 == BLUE && c2 == VIOLET) ||
                    (c1 == VIOLET && c2 == BLUE) -> INDIGO
            else -> throw Exception("Dirty Color")
        }

fun main(args: Array<String>) {
    println(mix(BLUE, YELLOW))
    println(mixOptimized(BLUE, YELLOW))
}

##2.17 重构之使用when替换if

interface Expr
class Num(val value: Int) : Expr
class Sum(val left: Expr, val right: Expr) : Expr

/*
fun eval(e: Expr) : Int {
    if (e is Num) {
        return e.value
    }

    if (e is Sum) {
        return eval(e.left) + eval(e.right)
    }

    throw IllegalArgumentException("Unknown expression")
}
*/

/*
fun eval(e: Expr) : Int =
    if (e is Num) {
        e.value
    } else if (e is Sum) {
        eval(e.left) + eval(e.right)
    } else {
        throw IllegalArgumentException("Unknown expression")
    }
*/

fun eval(e: Expr) : Int =
        when (e) {
            is Num ->
                e.value
            is Sum ->
                eval(e.left) + eval(e.right)
            else ->
                throw IllegalArgumentException("Unknown expression")
        }

fun main(args: Array<String>) {
    println(eval(Sum(Sum(Num(1), Num(2)), Num(4))))
}

##2.20 使用for-in循环遍历范围

fun divideInt(i: Int) = when {
i % 3 == 0 -> “$i: 3的倍数”
i % 4 == 0 -> “$i: 4的倍数”
else -> “$i”
}

fun main(args: Array) {
// 使用两个点来表示一个范围，不仅适用于数字，还适用于字母
val oneToTen = 1..10

// 正序遍历
for (i in oneToTen) {
    println(divideInt(i))
}

// 倒序遍历
for (i in 10 downTo 1) {
    println(divideInt(i))
}

// 指定步长  结果为10/8/6/4/2
for (i in 10 downTo 1 step 2) {
    println(divideInt(i))
}

// 不包含11的遍历
for (i in 1 until 11) {
    println(divideInt(i))
}

}

##2.22 使用in判断是否属于集合或范围

fun isLetter(c: Char) = c in 'a'..'z' || c in 'A'..'Z'
fun isNotDigit(c: Char) = c !in '0'..'9'

fun recognize(c: Char) = when (c) {
    in '0'..'9' -> "这是个数字！"
    in 'a'..'z', in 'A'..'Z' -> "这是个字母！"
    else -> "不知道..."
}

fun main(args: Array<String>) {
    println(isLetter('m'))    // true
    println(isNotDigit('x'))    // 这是个数字！

    println(recognize('8')) // true

    println("JavaScript" in "Java".."Kotlin") // true

    println("JavaScript" in setOf("Java", "Kotlin")) // false
    println("JavaScript" !in setOf("Java", "Kotlin")) // true
}

##2.24 try-catch-finally

import java.io.BufferedReader
import java.io.StringReader

fun readNumber(reader: BufferedReader): Int? {
    try {
        val line = reader.readLine()
        return Integer.parseInt(line)
    } catch (e: NumberFormatException) {
        return null
    } finally {
        reader.close()
    }
}

fun main(args: Array<String>) {
    // val reader = BufferedReader(StringReader("239"))   // 239
    val reader = BufferedReader(StringReader("abc"))      // null
    println(readNumber(reader))
}

##3.2 自定义函数以打印集合中的所有元素

fun main(args: Array<String>) {
    val list = listOf(1, 2, 3)
    println(list)   // [1, 2, 3]
    println(joinToString(list, "; ","(", ")"))      // (1; 2; 3)
}

fun <T> joinToString(collection: Collection<T>,
                     separator: String,
                     prefix: String,
                     postfix: String): String {
    val result = StringBuffer(prefix)

    for ((index, element) in collection.withIndex()) {
        if (index > 0) result.append(separator)
        result.append(element)
    }

    result.append(postfix)
    return result.toString()
}

##3.3 函数声明时的默认参数值

                     separator: String = ", ",

    for ((index, element) in collection.withIndex()) {
        if (index > 0) result.append(separator)
        result.append(element)

fun main(args: Array<String>) {
